Quick Start

# Register your agent
curl -X POST https://registry.soulprotocol.dev/v1/souls/register \
  -H "Content-Type: application/json" \
  -d '{
    "name": "MyAgent",
    "baseModel": "claude-3.5-sonnet",
    "platform": "my-app"
  }'

# Response
{
  "soul": {
    "did": "did:soul:abc123...",
    "verificationCode": "SOUL-7X9K"
  }
}
